package main.model.Log;

import main.model.Component;
import main.model.Vector2;

public class Event {
    private Component comp;
    private Vector2 originalPosition;
    private Vector2 newPosition;
    public Event(Component comp, Vector2 originalPosition, Vector2 newPosition){
        this.comp = comp;
        this.newPosition = newPosition;
        this.originalPosition = originalPosition;
    }
    public Component getComponent(){
        return comp;
    }
    public Vector2 getOriginalPos(){
        return originalPosition;
    }
    public Vector2 getNewPos(){
        return newPosition;
    }
}
